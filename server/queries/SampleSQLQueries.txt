District.Name,
DistrictAchievement.PerfIndexStore
FROM District
JOIN DistrictAchievement
ON District.DistrictIRN = DistrictAchievement.DistrictIRN
WHERE DistrictAchievement.PerfIndexStore > 70;

Building.BuildingIRN,
Building.Name,
BuildingEnrollment,
Building.AttendanceRate,
Building.MobilityRate,
Building.ChronicAbsenteeismRate,
BuildingAchievement.PerfIndexScore,

ON Building.BuildingIRN = BuildingAchievement.BuildingIRN
WHERE Building.Name = 'Asbury Elementary School';
